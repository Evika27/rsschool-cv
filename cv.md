_Write a function that reverses a string. The input string is given as an array of characters s._

_Input: s = ["h","e","l","l","o"]_
_Output: ["o","l","l","e","h"]_

```
const reverseString = function(s) {
    for(let i = 0; i < s.length/2; i++){
    let t = s[i];
    s[i] = s[s.length - i - 1];
    s[s.length - i - 1] = t;
    }
};
```
